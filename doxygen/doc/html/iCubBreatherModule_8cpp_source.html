113  bool respond(const Bottle &command, Bottle &reply)
114  {
115  int ack =Vocab::encode("ack");
116  int nack=Vocab::encode("nack");
117 
118  if (command.size()>0)
119  {
120  switch (command.get(0).asVocab())
121  {
122  case createVocab('s','t','a','r'):
123  {
124  int res=Vocab::encode("started");
125  if (iCubBreatherThrd -> startBreathing())
126  {
127  reply.addVocab(ack);
128  }
129  else
130  reply.addVocab(nack);
131 
132  reply.addVocab(res);
133  return true;
134  }
135  case createVocab('s','t','o','p'):
136  {
137  int res=Vocab::encode("stopped");
138  if (iCubBreatherThrd -> stopBreathing())
139  {
140  reply.addVocab(ack);
141  }
142  else
143  reply.addVocab(nack);
144 
145  reply.addVocab(res);
146  return true;
147  }
148  //-----------------
149  default:
150  return RFModule::respond(command,reply);
151  }
152  }
153 
154  reply.addVocab(nack);
155  return true;
156  }
